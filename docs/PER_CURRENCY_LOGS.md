for currency, logs in by_currency.items():
    for log in logs:

            logger.log_buy(
                currency=currency,
                volume=log.get('volume', 0),
                price=log.get('price', 0),
                delta_percent=log.get('delta_percent', 0),
                total_drop_percent=log.get('total_drop_percent', 0),
                investment=log.get('investment', 0)
            )
        elif log.get('type') == 'sell':
            logger.log_sell(
                currency=currency,
                volume=log.get('volume', 0),
                price=log.get('price', 0),
                delta_percent=log.get('delta_percent', 0),
                pnl=log.get('pnl', 0)
            )
